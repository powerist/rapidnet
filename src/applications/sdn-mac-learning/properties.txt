Learning Bridge

1. (@ Switch): Forall: have routing entry for MAC address A -> 
	(@Switch) have received a frame sourced from A 
	OR (@Nei) have forwarded a frame source from A
(Assumptions: no spoofing, no changing it to a frame that has source MAC address A)

forall Switch, Mac_A, OutPort, flowEntry(Switch, MAC_A, OutPort) ->
	exists Nei, DstMac, 
		packet(Switch, Nei, Mac_A, DstMac) **received packet from A** OR
		packet(Nei, Switch, Mac_A, DstMac) **forward packet sourced from from A**


2. Forall: (@End-host) Have received a frame B that is not destined to end-host A 
        -> (@Switch) Switch does not have a routing entry for B's dest address

forall Switch, Nei, Mac_A, Mac_B, Outport, packet(Switch, Nei, Mac_A, Mac_B) AND (Mac_A != Mac_B) ->
	forall Mac, (Mac != Mac_B) AND ((flowEntry(Switch, Mac, Outport) != flowEntry(Switch, Mac_B, OutPort))

3. (@Switch) Not exist a routing entry to MAC address A -> (@Switch) never have received/forwarded a packet sourced from A
      
forall Switch, Mac_A, OutPort, Nei, DstMac, Mac,
	(Mac != Mac_A) AND (flowEntry(Switch, Mac, Outport) != flowEntry(@Switch, Mac_A, OutPort))
	->
	(packet(Switch, Nei, Mac, DstMac) != packet(@Switch, Nei, Mac_A, DstMac)) **received packet from A** AND
	(packet(@Nei, Switch, Mac_A, DstMac) != packet(@Nei, Switch, Mac_A, DstMac)) **forward packet sourced from from A**
