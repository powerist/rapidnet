Learning Bridge

1. @ Sw: Forall: have routing entry for MAC address A -> have received/forwarded a frame sourced from A 
(with assumptions, no spoofing; or change it to a frame that has source MAC address A)

forall Switch Mac_A OutPort (
	flowEntry(@Switch, MAC_A, OutPort)

	->

	exists Nei DstMac
		packet(@Switch, Nei, Mac_A, DstMac) //received packet from A
		OR
		packet(@Nei, Switch, Mac_A, DstMac) //forward packet sourced from from A
)


2. @ End-host: Forall: Have received a frame B that is not destined to end-host A -> SW does not have a routing entry for B's dest address

forall Switch Nei Mac_A Mac_B Outport
	packet(@Switch, Nei, Mac_A, Mac_B)

	->

	not flowEntry(@Switch, Mac_B, OutPort)

3. @ SW: Not exist a routing entry to MAC address A -> never have received/forwarded a packet sourced from A

forall Switch Mac_A OutPort Nei DstMac (
	not flowEntry(@Switch, MAC_A, OutPort)

	->

	not packet(@Switch, Nei, Mac_A, DstMac) //received packet from A
	AND
	not packet(@Nei, Switch, Mac_A, DstMac) //forward packet sourced from from A
)
